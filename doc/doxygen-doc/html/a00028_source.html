00075 static inline int
00076 hwloc_cuda_get_device_cpuset(hwloc_topology_t topology ,
00077                              CUdevice cudevice, hwloc_cpuset_t set)
00078 {
00079 #ifdef HWLOC_LINUX_SYS
00080   /* If we're on Linux, use the sysfs mechanism to get the local cpus */
00081 #define HWLOC_CUDA_DEVICE_SYSFS_PATH_MAX 128
00082   char path[HWLOC_CUDA_DEVICE_SYSFS_PATH_MAX];
00083   FILE *sysfile = NULL;
00084   int domainid, busid, deviceid;
00085 
00086   if (hwloc_cuda_get_device_pci_ids(topology, cudevice, &domainid, &busid, &deviceid))
00087     return -1;
00088 
00089   sprintf(path, "/sys/bus/pci/devices/%04x:%02x:%02x.0/local_cpus", domainid, busid, deviceid);
00090   sysfile = fopen(path, "r");
00091   if (!sysfile)
00092     return -1;
00093 
00094   hwloc_linux_parse_cpumap_file(sysfile, set);
00095   if (hwloc_bitmap_iszero(set))
00096     hwloc_bitmap_copy(set, hwloc_topology_get_complete_cpuset(topology));
00097 
00098   fclose(sysfile);
00099 #else
00100   /* Non-Linux systems simply get a full cpuset */
00101   hwloc_bitmap_copy(set, hwloc_topology_get_complete_cpuset(topology));
00102 #endif
00103   return 0;
00104 }
