00664 enum hwloc_topology_flags_e {
00665   HWLOC_TOPOLOGY_FLAG_WHOLE_SYSTEM = (1<<0),
00674   HWLOC_TOPOLOGY_FLAG_IS_THISSYSTEM = (1<<1),
00694   /* \brief Detect PCI devices.
00695    *
00696    * By default, I/O devices are ignored. This flag enables I/O device
00697    * detection using the libpci backend. Only the common PCI devices (GPUs,
00698    * NICs, block devices, ...) and host bridges (objects that connect the host
00699    * objects to an I/O subsystem) will be added to the topology.
00700    * Uncommon devices and other bridges (such as PCI-to-PCI bridges) will be
00701    * ignored.
00702    */
00703   HWLOC_TOPOLOGY_FLAG_IO_DEVICES = (1<<2),
00704 
00705   /* \brief Detect PCI bridges.
00706    *
00707    * This flag should be combined with HWLOC_TOPOLOGY_FLAG_IO_DEVICES to enable
00708    * the detection of both common devices and of all useful bridges (bridges that
00709    * have at least one device behind them).
00710    */
00711   HWLOC_TOPOLOGY_FLAG_IO_BRIDGES = (1<<3),
00712 
00713   /* \brief Detect the whole PCI hierarchy.
00714    *
00715    * This flag enables detection of all I/O devices (even the uncommon ones)
00716    * and bridges (even those that have no device behind them) using the libpci
00717    * backend.
00718    */
00719   HWLOC_TOPOLOGY_FLAG_WHOLE_IO = (1<<4)
00720 };
